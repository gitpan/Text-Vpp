@PERL << ^END_OF_PERL$
  sub Hello {
    Vpp_Out(join('',"Hello world"," - Here I am"),"and here on the next line");
  }
  
  sub Chances($$) {
    my ($nseq,$num) = @_;
    # compute the chance to draw a sequence of <nseq> specific balls
    # out of <num> balls.
    my $chance;
    if ( $nseq > $num ) {
      $chance= 0;
    } else {
      $chance= 1;
      for (my $k=1; $k <= $nseq; $k++) {
        $chance*= $k/($num-$k);
      }
    }
    Vpp_Out("double chance_${nseq}_of_$num = $chance;");
  }
  
END_OF_PERL
